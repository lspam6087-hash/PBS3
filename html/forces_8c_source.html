9// This function calculates all forces acting on the particles (bonded and non-bonded).
10// It initializes the forces array, then calculates bond-stretch, angle-bend, dihedral-torsion,
11// and non-bonded forces. The total potential energy is returned.
12double calculate_forces(struct Parameters *p_parameters, struct Nbrlist *p_nbrlist, struct Vectors *p_vectors)
13{
14 struct Vec3D *f = p_vectors->f;
15 size_t num_part = p_parameters->num_part;
16
17 // Initialize the forces to zero for all particles
18 for (size_t i = 0; i < num_part; i++)
19 f[i] = (struct Vec3D){0.0, 0.0, 0.0};
20
21 // Calculate the forces and accumulate the potential energy from each type of interaction
22 double Epot = calculate_forces_bond(p_parameters, p_vectors);
23 Epot += calculate_forces_angle(p_parameters, p_vectors);
24 Epot += calculate_forces_dihedral(p_parameters, p_vectors);
25 Epot += calculate_forces_nb(p_parameters, p_nbrlist, p_vectors);
26
27 return Epot;
28}

30// This function calculates bond-stretch forces based on the current positions of the bonded particles.
31// It applies the minimum image convention to calculate the distance between bonded pairs and then
32// computes the force and potential energy due to the bond interaction.
33double calculate_forces_bond(struct Parameters *p_parameters, struct Vectors *p_vectors)
34{
35 double Epot = 0.0;
36 struct Bond *bonds = p_vectors->bonds;
37 size_t num_bonds = p_vectors->num_bonds;
38 struct Vec3D *f = p_vectors->f;
39 struct Vec3D *r = p_vectors->r;
40 struct Vec3D L = p_parameters->L;
41 struct Vec3D rij;
42 struct Vec3D fi = {0.0, 0.0, 0.0};
43
44 // Loop through each bond and calculate the forces
45 for (size_t q = 0; q < num_bonds; ++q)
46 {
47 size_t i = bonds[q].i;
48 size_t j = bonds[q].j;
49
50 // Apply the minimum image convention for calculating distances
51 rij.x = r[i].x - r[j].x;
52 rij.x = rij.x - L.x * floor(rij.x / L.x + 0.5);
53 rij.y = r[i].y - r[j].y;
54 rij.y = rij.y - L.y * floor(rij.y / L.y + 0.5);
55 rij.z = r[i].z - r[j].z;
56 rij.z = rij.z - L.z * floor(rij.z / L.z + 0.5);
57
59
60 f[i].x += fi.x;
61 f[i].y += fi.y;
62 f[i].z += fi.z;
63 f[j].x -= fi.x;
64 f[j].y -= fi.y;
65 f[j].z -= fi.z;
66 }
67
68 return Epot; // Return the potential energy due to bond-stretch interactions
69}

174// This function calculates non-bonded forces between particles using the neighbor list.
175// The potential energy and forces are calculated using the Lennard-Jones potential.
176double calculate_forces_nb(struct Parameters *p_parameters, struct Nbrlist *p_nbrlist, struct Vectors *p_vectors)
177{
178 struct Vec3D df;
179 double r_cutsq, sigmasq, sr2, sr6, sr12, fr;
180 struct DeltaR rij;
181 struct Pair *nbr = p_nbrlist->nbr;
182 const size_t num_nbrs = p_nbrlist->num_nbrs;
183 struct Vec3D *f = p_vectors->f;
184 size_t num_part = p_parameters->num_part;
185
186 r_cutsq = p_parameters->r_cut * p_parameters->r_cut;
187 sigmasq = p_parameters->sigma * p_parameters->sigma;
188 double epsilon = p_parameters->epsilon;
189
190 double Epot = 0.0, Epot_cutoff;
191 sr2 = sigmasq / r_cutsq;
192 sr6 = sr2 * sr2 * sr2;
193 sr12 = sr6 * sr6;
194 Epot_cutoff = sr12 - sr6;
195
196 // Loop through the neighbor list and calculate the forces for each particle pair
197 for (size_t k = 0; k < num_nbrs; k++)
198 {
199 rij = nbr[k].rij;
200 size_t i = nbr[k].i;
201 size_t j = nbr[k].j;
202
203 // Compute forces if the distance is smaller than the cutoff distance
204 if (rij.sq < r_cutsq)
205 {
207
208 sr2 = sigmasq / rij.sq;
209 sr6 = sr2 * sr2 * sr2;
210 sr12 = sr6 * sr6;
211
212 // Calculate the potential energy
213 Epot += 4.0 * epsilon * (sr12 - sr6 - Epot_cutoff);
214
215 // Compute the force and apply it to both particles
216 fr = 24.0 * epsilon * (2.0 * sr12 - sr6) / rij.sq; // Force divided by distance
217 df.x = fr * rij.x;
218 df.y = fr * rij.y;
219 df.z = fr * rij.z;
220
221 f[i].x += df.x;
222 f[i].y += df.y;
223 f[i].z += df.z;
224 f[j].x -= df.x;
225 f[j].y -= df.y;
226 f[j].z -= df.z;
227 }
228 }
229
230 return Epot; // Return the potential energy due to non-bonded interactions
231}
