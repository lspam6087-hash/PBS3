220// This function initializes the velocities of particles based on the Maxwell-Boltzmann distribution.
221// The total momentum is also removed to ensure zero total momentum (important for stability).
222void initialise_velocities(struct Parameters *p_parameters, struct Vectors *p_vectors)
223{
224 double sqrtktm = sqrt(p_parameters->kT / p_parameters->mass);
225 struct Vec3D sumv = {0.0, 0.0, 0.0}; // Total velocity (to remove later)
226
227 // Assign random velocities to each particle
228 for (size_t i = 0; i < p_parameters->num_part; i++)
229 {
230 p_vectors->v[i].x = sqrtktm * gauss();
231 p_vectors->v[i].y = sqrtktm * gauss();
232 p_vectors->v[i].z = sqrtktm * gauss();
233 sumv.x += p_vectors->v[i].x;
234 sumv.y += p_vectors->v[i].y;
235 sumv.z += p_vectors->v[i].z;
236 }
237
238 // Remove the average velocity to ensure zero total momentum
239 sumv.x /= ((double)(p_parameters->num_part));
240 sumv.y /= ((double)(p_parameters->num_part));
241 sumv.z /= ((double)(p_parameters->num_part));
242 for (size_t i = 0; i < p_parameters->num_part; i++)
243 {
244 p_vectors->v[i].x -= sumv.x;
