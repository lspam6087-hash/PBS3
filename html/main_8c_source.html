55int main(void)
56{
57 struct Vectors vectors;
58 struct Parameters parameters;
59 struct Nbrlist nbrlist;
60 size_t step;
61 double Ekin, Epot, time;
62
63 // Step 1: Set the simulation parameters from input files
64 set_parameters(&parameters);
65
66 // Step 2: Allocate memory for particles, forces, and neighbor lists
67 alloc_memory(&parameters, &vectors, &nbrlist);
68
69 // Check if a restart is required
70 if (parameters.load_restart == 1)
71 {
72 load_restart(&parameters, &vectors);
73 initialise_structure(&parameters, &vectors, &nbrlist);
74 step = 0;
75 time = 0.0;
76 }
77 else
78 {
81
82 initialise(&parameters, &vectors, &nbrlist, &step, &time);
83 }
84
85 // Step 3: Build the neighbor list for non-bonded interactions
86 build_nbrlist(&parameters, &vectors, &nbrlist);
87
88 // Step 4: Calculate initial forces (non-bonded and bonded if implemented)
89 Epot = calculate_forces(&parameters, &nbrlist, &vectors);
90
91 // Output initial particle positions in PDB format
92 record_trajectories_pdb(1, &parameters, &vectors, time);
93
94 // Main MD loop using velocity-Verlet integration
95 while (step < parameters.num_dt_steps)
96 {
97 step++;
98 time += parameters.dt;
99
100 // Update velocities (half-step)
102 Ekin = update_velocities_half_dt(&parameters, &nbrlist, &vectors);
103
105 thermostat(&parameters, &vectors, Ekin);
106
107 // Update positions
108 update_positions(&parameters, &nbrlist, &vectors);
109
110 // Apply boundary conditions
111 boundary_conditions(&parameters, &vectors);
112
113 // Rebuild neighbor list if needed
114 update_nbrlist(&parameters, &vectors, &nbrlist);
115
116 // Calculate forces for the current configuration (bonded forces if implemented)
117 Epot = calculate_forces(&parameters, &nbrlist, &vectors);
118
119 // Final velocity update (half-step)
120 Ekin = update_velocities_half_dt(&parameters, &nbrlist, &vectors);
121
122 // Output system state every 'num_dt_pdb' steps
123 if (step % parameters.num_dt_pdb == 0)
124 record_trajectories_pdb(0, &parameters, &vectors, time);
125
126 // Save restart file every 'num_dt_restart' steps
127 if (step % parameters.num_dt_restart == 0)
128 save_restart(&parameters, &vectors);
129
131
132 // Print to the screen to monitor the progress of the simulation
134 printf("Step %lu, Time %f, Epot %f, Ekin %f, Etot %f\n", (long unsigned)step, time, Epot, Ekin, Epot + Ekin);
135 }
136
137 // Save final state
138 save_restart(&parameters, &vectors);
139
140 // Step 5: Free memory and clean up
141 free_memory(&vectors, &nbrlist);
142
143 return 0;
144}
