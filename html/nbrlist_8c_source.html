9// Allocate memory for the cell-linked list based on the number of particles and grid size.
10// The grid size is determined by dividing the simulation box into cells, each larger than the interaction range.
11void alloc_celllist(struct Parameters *p_parameters, struct Celllist *p_celllist)
12{
13 struct Index3D size_grid;
14 size_t Mtot;
15 const double rlist = p_parameters->r_cut + p_parameters->r_shell;
16 const size_t num_part = p_parameters->num_part;
17
18 // Determine the size of the grid based on cutoff distance and box dimensions
19 size_grid.i = floor(p_parameters->L.x / rlist);
20 size_grid.j = floor(p_parameters->L.y / rlist);
21 size_grid.k = floor(p_parameters->L.z / rlist);
22 p_celllist->size_grid.i = size_grid.i;
23 p_celllist->size_grid.j = size_grid.j;
24 p_celllist->size_grid.k = size_grid.k;
25
26 // Calculate total number of cells and allocate memory for the cell lists
27 Mtot = size_grid.i * size_grid.j * size_grid.k;
28 p_celllist->head = (size_t *)malloc(Mtot * sizeof(size_t));
29 p_celllist->num_cells_max = Mtot;
30 p_celllist->num_cells = Mtot;
31 p_celllist->num_part_max = num_part;
32 p_celllist->particle2cell = (size_t *)malloc(num_part * sizeof(size_t));
33 p_celllist->list = (size_t *)malloc(num_part * sizeof(size_t));
34}

47// Build the cell-linked list by assigning particles to grid cells based on their positions.
48// This is the first step before building the neighbor list.
49void build_celllist(struct Parameters *p_parameters, struct Vectors *p_vectors, struct Celllist *p_celllist)
50{
51 struct Index3D size_grid, indx;
52 size_t Mtot, icell;
53 const double rlist = p_parameters->r_cut + p_parameters->r_shell;
54 struct Vec3D mL;
55 const size_t num_part = p_parameters->num_part;
56 size_t *particle2cell, *head, *celllist;
57 struct Vec3D *r;
58
59 size_grid.i = floor(p_parameters->L.x / rlist);
60 size_grid.j = floor(p_parameters->L.y / rlist);
61 size_grid.k = floor(p_parameters->L.z / rlist);
62 p_celllist->size_grid.i = size_grid.i;
63 p_celllist->size_grid.j = size_grid.j;
64 p_celllist->size_grid.k = size_grid.k;
65 Mtot = size_grid.i * size_grid.j * size_grid.k;
66 if (Mtot > p_celllist->num_cells_max)
67 {
68 p_celllist->head = (size_t *)realloc(p_celllist->head, Mtot * sizeof(size_t));
69 p_celllist->num_cells_max = Mtot;
70 }
71 if (num_part > p_celllist->num_part_max)
72 {
73 p_celllist->particle2cell = (size_t *)realloc(p_celllist->particle2cell, num_part * sizeof(size_t));
74 p_celllist->list = (size_t *)realloc(p_celllist->list, num_part * sizeof(size_t));
75 p_celllist->num_part_max = num_part;
76 }
77 mL.x = ((double)size_grid.i) / p_parameters->L.x;
78 mL.y = ((double)size_grid.j) / p_parameters->L.y;
79 mL.z = ((double)size_grid.k) / p_parameters->L.z;
80 head = p_celllist->head;
81 for (icell = 0; icell < Mtot; ++icell)
82 head[icell] = SIZE_MAX;
83 particle2cell = p_celllist->particle2cell;
84 celllist = p_celllist->list;
85 r = p_vectors->r;
86 for (size_t i = 0; i < num_part; ++i)
87 {
88 indx.i = floor(r[i].x * mL.x);
89 indx.j = floor(r[i].y * mL.y);
90 indx.k = floor(r[i].z * mL.z);
91 icell = indx.i + size_grid.i * (indx.j + indx.k * size_grid.j);
92 particle2cell[i] = icell;
93 celllist[i] = head[icell];
94 head[icell] = i;
95 }
96}

98// Allocate memory for the neighbor list. This includes arrays for storing neighbors and bonded interactions.
99void alloc_nbrlist(struct Parameters *p_parameters, struct Nbrlist *p_nbrlist)
100{
101 double Ndouble = (double)p_parameters->num_part;
102 double rlist = p_parameters->r_cut + p_parameters->r_shell;
103 // Estimate the number of neighbors using the average density
104 double Nnbr = 0.5 * Ndouble * (4.0 / 3.0 * PI * rlist * rlist * rlist) / (p_parameters->L.x * p_parameters->L.y * p_parameters->L.z);
105 size_t num_nbrs_max = ceil(0.6 * Ndouble * (Nnbr + 2.0 * sqrt(Nnbr)));
106 p_nbrlist->p_celllist = (struct Celllist *)malloc(sizeof(struct Celllist));
107 alloc_celllist(p_parameters, p_nbrlist->p_celllist);
108 p_nbrlist->num_nbrs_max = num_nbrs_max;
109 p_nbrlist->nbr = (struct Pair *)malloc(num_nbrs_max * sizeof(struct Pair));
110 p_nbrlist->dr = (struct DeltaR *)malloc(p_parameters->num_part * sizeof(struct DeltaR));
111 p_nbrlist->head12 = NULL;
112 p_nbrlist->pairs12 = NULL;
113 p_nbrlist->head13 = NULL;
114 p_nbrlist->pairs13 = NULL;
115 p_nbrlist->head14 = NULL;
116 p_nbrlist->pairs14 = NULL;
117}

176// Build the neighbor list by checking all pairs of particles within a given cutoff distance.
177// Exclude bonded interactions (1-2, 1-3) if necessary.

184 const double rlist_sq = rlist * rlist;

213 {

217 rij.y = ri.y - r[j].y;

311// This function updates the neighbor list by recalculating the positions of particles relative to
312// the positions when the neighbor list was last built. If a particle has moved far enough (beyond
313// the skin distance), the neighbor list is rebuilt.
314int update_nbrlist(struct Parameters *p_parameters, struct Vectors *p_vectors, struct Nbrlist *p_nbrlist)
315{
316 const double dr_sq_max = 0.25 * (p_parameters->r_shell * p_parameters->r_shell);
317 int isRebuild = 0;
318 struct DeltaR rij;
319 struct Pair *nbr = p_nbrlist->nbr;
320 struct Vec3D *dr = p_vectors->dr;
321 // The neighbor list needs to be rebuild if one of the particles has displaced more then 0.5*r_shell
322 for (size_t i = 0; i < p_parameters->num_part; ++i)
323 if ((p_nbrlist->dr[i].sq) > dr_sq_max)
324 {
325 isRebuild = 1;
326 }
327 if (isRebuild) // rebuild neighbor list
328 build_nbrlist(p_parameters, p_vectors, p_nbrlist);
329 else // If no rebuild is needed, update the values of the connecting vectors
330 {
331 for (size_t k = 0; k < p_nbrlist->num_nbrs; ++k)
332 {
333 size_t i = nbr[k].i;
334 size_t j = nbr[k].j;
335 rij = nbr[k].rij;
336 rij.x += (dr[i].x - dr[j].x);
337 rij.y += (dr[i].y - dr[j].y);
338 rij.z += (dr[i].z - dr[j].z);
339 rij.sq = rij.x * rij.x + rij.y * rij.y + rij.z * rij.z;
340 p_nbrlist->nbr[k].rij = rij;
341 }
342 }
343 return isRebuild;
344}
